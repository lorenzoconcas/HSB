118 public static Encoding GetEncoding(byte[] file)
119 {
120 var bom = new byte[4];
121 bom[0] = file[0];
122 bom[1] = file[1];
123 bom[2] = file[2];
124 bom[3] = file[3];
125
126 // Analyze the BOM
127
128 // if (bom[0] == 0x2b && bom[1] == 0x2f && bom[2] == 0x76) return Encoding.UTF7;
129 if (bom[0] == 0xef && bom[1] == 0xbb && bom[2] == 0xbf) return Encoding.UTF8;
130 if (bom[0] == 0xff && bom[1] == 0xfe && bom[2] == 0 && bom[3] == 0) return Encoding.UTF32; //UTF-32LE
131 if (bom[0] == 0xff && bom[1] == 0xfe) return Encoding.Unicode; //UTF-16LE
132 if (bom[0] == 0xfe && bom[1] == 0xff) return Encoding.BigEndianUnicode; //UTF-16BE
133 if (bom[0] == 0 && bom[1] == 0 && bom[2] == 0xfe && bom[3] == 0xff) return new UTF32Encoding(true, true); //UTF-32BE
134
135
136
137 // We actually have no idea what the encoding is if we reach this point, so
138 // you may wish to return null instead of defaulting to ASCII
139 return Encoding.ASCII;
140 }
