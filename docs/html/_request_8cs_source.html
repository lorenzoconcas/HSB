46 {

48 rawData = data;
49 rawBody = [];
50 this.config = config;
51 requestContent = [];
52 IsTLS = isTLS;
53
54 if (data == null || data.Length == 0)

56 return;
57 }

59 //extract ipv4 or ipv6 from the remote endpoint
60 if (rEP != null)
61 {
62 var rIEP = (IPEndPoint)rEP;
63 clientIP = rIEP.Address.ToString();
64 clientPort = rIEP.Port;

68 switch (Utils.GetEncoding(data))
69 {
70 case UTF8Encoding:

73 case UTF32Encoding:
74 reqText = Encoding.UTF32.GetString(data);
75 break;
76 case ASCIIEncoding:

81 if (reqText.Replace("\0", "") == "")
82 {
83 //note:
84 //it can happen in programs like postman that a request to localhost produces two requests
85 //one for IPv6 and one for IPv4
86 //i don't know why but the second request is invalid
87 validRequest = false;
88 config.Debug.INFO("Got an invalid request, ignoring...");
89 requestContent.Add(" ");
90 return;
91 }
92 // reqText = Encoding.UTF8.GetString(data);

99 private void ParseRequest()
100 {
101 if (reqText == "")
102 {
103 //empty request
104 _url = "/";
105 _protocol = HTTP_PROTOCOL.HTTP1_0;
106 _method = HTTP_METHOD.GET;
107 body = "";
108 rawBody = [];
109 session = new Session(); //default, invalid session
110 Terminal.INFO("Got an empty request, setting default values");
111 return;
112 }
113
114 try
115 {
116 string[] firstLine = requestContent.First().Split(" ");
117 _method = HttpUtils.GetMethod(firstLine[0]);
118 _url = firstLine[1].Split("?")[0];
119 _protocol = HttpUtils.GetProtocol(firstLine[2]);
120
121 //collect parameters
122 if (firstLine[1].Replace(_url, "") != "")
123 {
124 List<string> prms = [.. firstLine[1].Split("?")[1].Split("&")];
125
126 foreach (string p in prms)
127 {
128 if (p != "" && !parameters.ContainsKey(p)) //skip empty parameters and no duplicates
129 parameters.Add(p.Split("=")[0], p.Split("=")[1]);
130 }
131 }
132
133
134 //collect headers
135 foreach (string r in requestContent)
136 {
137 //skip if first element
138 if (r == requestContent.First())
139 continue;
140 if (r == "")
141 {
142 break;
143 }
144
145 rawHeaders.Add(r);
146 string[] header = r.Split(": ");
147 if (header.Length == 2)
148 headers.Add(header[0], header[1]);
149 else headers.Add(r, "");
150
151 }
152
153 //auth data collection
154 //basic auth
155 if (headers.TryGetValue("Authorization", out string? value))
156 {
157 var _auth = value;
158 try
159 {
160 _auth = Encoding.UTF8.GetString(Convert.FromBase64String(_auth));
161 var x = _auth.Split(":");
162 if (_auth.Length == 2)
163 {
164 basicAuth = new(x[0], x[1]);
165 }
166
167 }
168 catch (Exception)
169 {
170 if (_auth.Contains("Bearer"))
171 {
172 oAuth2_0Token = value;
173 }
174 }
175 }
176
177 //oAuth1.0 information
178 TryExtractAndSetOAuth1_0();
179
180 //oAuth2.0 token
181 if (parameters.TryGetValue("access_token", out string? tkn))
182 {
183 oAuth2_0Token = tkn;
184 }
185
186
187 //parse cookies
188 if (headers.TryGetValue("Cookie", out string? val))
189 {
190
191 var strings = val.Split("; ");
192 foreach (var s in strings)
193 {
194
195 cookies.Add(s.Split("=")[0], new Cookie(s));
196 }
197 }
198
199 //search for a session token
200 if (cookies.TryGetValue("hsbst", out Cookie? cookie) && SessionManager.GetInstance().IsValidSession(cookie.value))
201 {
202 session = SessionManager.GetInstance().GetSession(cookie.value);
203 }
204 else
205 {
206
207 session = new()
208 {
209 ExpirationTime = DateTime.Now.AddTicks((long)config.DefaultSessionExpirationTime).Ticks
210 };
211 string sessionToken = SessionManager.GetInstance().CreateSession(session);
212
213 Cookie c = new()
214 {
215 name = "hsbst",
216 value = sessionToken,
217 expiration = DateTime.Now.AddTicks((long)config.DefaultSessionExpirationTime),
218 path = "/",
219 priority = Cookie.CookiePriority.HIGH
220 };
221
222 config.AddCustomGlobalCookie(c);
223 }
224
225 //extract body, which is the remaining part of the request text
226
227 int offset = Utils.IndexOf(rawData, "\r\n\r\n"u8.ToArray()) + 4;
228 rawBody = rawData[offset..];
229 body = Encoding.UTF8.GetString(rawBody);
230
231 if (IsFileUpload())
232 {
233 multiPartFormData = new MultiPartFormData(rawBody, headers["Content-Type"].Split("boundary=")[1]);
234 }
235 if (IsFormUpload())
236 {
237 form = new Form(body);
238 }
239
240 validRequest = true;
241
242 }
243 catch (Exception e)
244 {
245 Terminal.WriteLine("Invalid request, reason : " + e.Message, BG_COLOR.BLACK, FG_COLOR.RED);
246 validRequest = false;
247 return;
248 }
249
250 }
