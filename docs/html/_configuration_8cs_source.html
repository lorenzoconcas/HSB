130 public Configuration(string content)

132 using var doc = JsonDocument.Parse(content);

134

137 Address = root.GetProperty("Address").GetString() ?? "";
138 Port = root.GetProperty("Port").GetInt16();
139 StaticFolderPath = root.GetProperty("StaticFolderPath").GetString() ?? "";
140 Debug = Debugger.FromJson(root.GetProperty("Debug"));
141 SslSettings = SslConfiguration.FromJSON(root.GetProperty("SslConfiguration"));
142 RequestMaxSize = root.GetProperty("Port").GetInt32();
143 BlockMode = (BLOCK_MODE)root.GetProperty("BlockMode").GetInt32();
144 HideBranding = root.GetProperty("HideBranding").GetBoolean();
145 IPAutoblock = root.GetProperty("IPAutoblock").GetBoolean();
146 ListeningMode = (IPMode)root.GetProperty(nameof(ListeningMode)).GetInt32();
147 CustomServerName = root.GetProperty("CustomServerName").GetString() ?? "";
148 ServeEmbeddedResource = root.GetProperty("ServeEmbeddedResource").GetBoolean();
149 EmbeddedResourcePrefix = root.GetProperty("EmbeddedResourcePrefix").GetString() ?? "";
150 DefaultSessionExpirationTime = root.GetProperty("DefaultSessionExpirationTime").GetUInt64();
151
152 foreach (var item in root.GetProperty("PermanentIPList").EnumerateArray())
153 {
154 string? v = item.GetString();
155 if (v != null)
156 {
157 PermanentIPList.Add(v!);
158 }
159 }

164 catch (Exception e)
165 {
166 Terminal.ERROR("Cannot parse configuration file");
167 Terminal.ERROR(e);
168 Environment.Exit(1);

350 public override string ToString()
351 {
352 string str = $"Current configuration:\nListening address and port: {Address}:{Port}";
353 if (StaticFolderPath == "")
354 str += "\nStatic folder is not set";
355 else
356 str += $"\nStatic folder path: {StaticFolderPath}";
357
358 if (expressMapping.Any())
359 {
360 str += "\nExpressJS-Like routing map:";
361 expressMapping.ForEach(m => str += $"\nPath : {m.Item1} -> {m.Item2.Item2.Method.Name}");
362 }
363
364 var staticRoutes = Server.CollectStaticRoutes();
365
366 if (staticRoutes.Any())
367 {
368 str += "\nStatic routes:";
369 staticRoutes.ToList().ForEach(m => str += $"\nPath : {m.Key.Item1} -> {m.Value.Name}");
370 }
371 return str;
372 }

374 public List<Tuple<string, string>> GetAllRoutes()
375 {
376 List<Tuple<string, string>> routes = new();
377 expressMapping.ForEach(m => routes.Add(new(m.Item1, m.Item2.Item1.ToString())));
378 var staticRoutes = Server.CollectStaticRoutes();
379 staticRoutes.ToList().ForEach(m => routes.Add(new(m.Key.Item1, "")));
380 return routes;
381 }
