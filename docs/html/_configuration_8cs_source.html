278 string str = $"Current configuration:\nListening address and port: {address}:{port}";
279 if (staticFolderPath == "")

281 else

283
284 if (expressMapping.Any())

286 str += "\nExpressJS-Like routing map:";
287 expressMapping.ForEach(m => str += $"\nPath : {m.Item1} -> {m.Item2.Item2.Method.Name}");
288 }
289
290 var staticRoutes = Server.CollectStaticRoutes();
291
292 if (staticRoutes.Any())
293 {
294 str += "\nStatic routes:";
295 staticRoutes.ToList().ForEach(m => str += $"\nPath : {m.Key.Item1} -> {m.Value.Name}");
296 }
297
298
