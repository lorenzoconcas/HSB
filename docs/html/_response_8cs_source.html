104 public void SendFile(string absPath, string? mimeType = null, int statusCode = 200, Dictionary<string, string>? customHeaders = null)
105 {
106 var data = File.ReadAllBytes(absPath);
107
108 string _mime = mimeType ?? MimeTypeUtils.GetMimeType(Path.GetExtension(absPath)) ?? MimeTypeUtils.APPLICATION_OCTET;
109 string headers = GetHeaders(statusCode, data.Length, _mime, customHeaders);
110 byte[] headersBytes = Encoding.UTF8.GetBytes(headers);
111 byte[] responseBytes = new byte[data.Length + headersBytes.Length];

113 headersBytes.CopyTo(responseBytes, 0);
114 data.CopyTo(responseBytes, headersBytes.Length);

276 private string GetHeaders(int responseCode, int size, string contentType, Dictionary<string, string>? customHeaders = null)
277 {
278 CultureInfo ci = new("en-US");
279
280 string currentTime = DateTime.Now.ToString("ddd, dd MMM yyy HH:mm:ss ", ci) + "GMT";
281
282 string headers = $"{HttpUtils.ProtocolAsString(request.PROTOCOL)} {responseCode} {request.URL} {NEW_LINE}";
283 headers += "Date: " + currentTime + NEW_LINE;
284
285 headers += $"Server : HSB-#/{Assembly.GetExecutingAssembly().GetName().Version} ({GetOSInfo()})" + NEW_LINE;
286 headers += $"Last-Modified: {currentTime}{NEW_LINE}";
287 headers += $"Content-Length: {size}{NEW_LINE}";
288 headers += $"Content-Type: {contentType}{NEW_LINE}";
289
290 if (customHeaders != null)
291 foreach (var h in customHeaders)
292 headers += $"{h.Key}: {h.Value}{NEW_LINE}";
293
294 if (config.CustomGlobalHeaders.Any())
295 {
296 foreach (var h in config.CustomGlobalHeaders)
297 {
298 headers += $"{h.Key}: {h.Value}{NEW_LINE}";
299 }
300 }
301
302 if (config.CustomGlobalCookies.Any())
303 {
304 foreach (var c in config.CustomGlobalCookies)
305 {
306 headers += $"Set-Cookie: {c.Value}{NEW_LINE}";
307 }
308 }
309
310 /* if (request.GetHeaders["Connection"] != null)
311 {
312 headers += $"Connection: {request.GetHeaders["Connection"]}";
313 }
314 else
315 {*/
316 //visit https://httpwg.org/specs/rfc9113.html#ConnectionSpecific, p8.2.2 (27-Jun-23)
317 if (request.PROTOCOL == HTTP_PROTOCOL.HTTP1_0 || request.PROTOCOL == HTTP_PROTOCOL.HTTP1_1)
318 headers += "Connection: Close";
319
320 //}
321 headers += NEW_LINE + NEW_LINE;
322 return headers;
323 }
