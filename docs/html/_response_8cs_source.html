119 public void SendFile(string absPath, string? mimeType = null, int statusCode = HTTP_CODES.OK, Dictionary<string, string>? customHeaders = null)
120 {
121 var data = File.ReadAllBytes(absPath);
122
123 string _mime = mimeType ?? MimeTypeUtils.GetMimeType(Path.GetExtension(absPath)) ?? MimeTypeUtils.APPLICATION_OCTET;
124 string headers = GetHeaders(statusCode, data.Length, _mime, customHeaders);
125 byte[] headersBytes = Encoding.UTF8.GetBytes(headers);
126 byte[] responseBytes = new byte[data.Length + headersBytes.Length];
127
128 headersBytes.CopyTo(responseBytes, 0);
129 data.CopyTo(responseBytes, headersBytes.Length);
130
131 Send(responseBytes);
132 }

317 private string GetHeaders(int responseCode, int size, string contentType, Dictionary<string, string>? customHeaders = null)
318 {
319 CultureInfo ci = new("en-US");
320
321 string currentTime = DateTime.Now.ToString("ddd, dd MMM yyy HH:mm:ss ", ci) + "GMT";
322
323 string headers = $"{HttpUtils.ProtocolAsString(request.PROTOCOL)} {responseCode} {request.URL} {NEW_LINE}";
324 headers += "Date: " + currentTime + NEW_LINE;
325 if (config.CustomServerName != "")
326 headers += $"Server: {config.CustomServerName}{NEW_LINE}";
327 else
328 headers += $"Server: HSB-#/{Assembly.GetExecutingAssembly().GetName().Version} ({GetOSInfo()})" + NEW_LINE;
329
330 headers += $"Last-Modified: {currentTime}{NEW_LINE}";
331
332
333 if (responseCode < 300 || responseCode > 399)
334 {
335 headers += $"Content-Length: {size}{NEW_LINE}";
336 headers += $"Content-Type: {contentType}{NEW_LINE}";
337 }
338
339 if (customHeaders != null)
340 {
341
342 foreach (var h in customHeaders)
343 headers += $"{h.Key}: {h.Value}{NEW_LINE}";
344
345 //if it's a redirect "Location" header is a must
346 if (responseCode >= 300 && responseCode <= 399 && !customHeaders.ContainsKey("Location"))
347 {
348 throw new InvalidRedirectRoute();
349 }
350 }
351
352 if (config.CustomGlobalHeaders.Any())
353 {
354 foreach (var h in config.CustomGlobalHeaders)
355 {
356 headers += $"{h.Key}: {h.Value}{NEW_LINE}";
357 }
358 }
359
360 if (config.CustomGlobalCookies.Any())
361 {
362 foreach (var c in config.CustomGlobalCookies)
363 {
364 headers += $"Set-Cookie: {c.Value}{NEW_LINE}";
365 }
366 }
367
368 /* if (request.GetHeaders["Connection"] != null)
369 {
370 headers += $"Connection: {request.GetHeaders["Connection"]}";
371 }
372 else
373 {*/
374 //visit https://httpwg.org/specs/rfc9113.html#ConnectionSpecific, p8.2.2 (27-Jun-23)
375 if (request.PROTOCOL == HTTP_PROTOCOL.HTTP1_0 || request.PROTOCOL == HTTP_PROTOCOL.HTTP1_1)
376 headers += "Connection: Close";
377
378 //}
379 headers += NEW_LINE + NEW_LINE;
380 return headers;
381 }
