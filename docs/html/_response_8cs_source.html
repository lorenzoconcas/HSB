122 public void SendFile(string absPath, string? mimeType = null, int statusCode = HTTP_CODES.OK, Dictionary<string, string>? customHeaders = null)
123 {
124 var data = File.ReadAllBytes(absPath);
125
126 string _mime = mimeType ?? MimeTypeUtils.GetMimeType(Path.GetExtension(absPath)) ?? MimeTypeUtils.APPLICATION_OCTET;
127 string headers = GetHeaders(statusCode, data.Length, _mime, customHeaders);
128 byte[] headersBytes = Encoding.UTF8.GetBytes(headers);
129 byte[] responseBytes = new byte[data.Length + headersBytes.Length];
130
131 headersBytes.CopyTo(responseBytes, 0);
132 data.CopyTo(responseBytes, headersBytes.Length);
133
134 Send(responseBytes);
135
136 //clear memory after sending
137 data = [];
138 headersBytes = [];
139 responseBytes = [];
140 }

329 private string GetHeaders(int responseCode, int size, string contentType, Dictionary<string, string>? customHeaders = null)
330 {
331 CultureInfo ci = new("en-US");
332
333 string currentTime = DateTime.Now.ToString("ddd, dd MMM yyy HH:mm:ss ", ci) + "GMT";
334
335 string headers = $"{HttpUtils.ProtocolAsString(request.PROTOCOL)} {responseCode} {request.URL} {NEW_LINE}";
336 headers += "Date: " + currentTime + NEW_LINE;
337 if (config.CustomServerName != "")
338 headers += $"Server: {config.CustomServerName}{NEW_LINE}";
339 else
340 headers += $"Server: HSB-#/{Assembly.GetExecutingAssembly().GetName().Version} ({GetOSInfo()})" + NEW_LINE;
341
342 headers += $"Last-Modified: {currentTime}{NEW_LINE}";
343
344
345 if (responseCode < 300 || responseCode > 399)
346 {
347 headers += $"Content-Length: {size}{NEW_LINE}";
348 headers += $"Content-Type: {contentType}{NEW_LINE}";
349 }
350
351 if (customHeaders != null)
352 {
353
354 foreach (var h in customHeaders)
355 headers += $"{h.Key}: {h.Value}{NEW_LINE}";
356
357 //if it's a redirect "Location" header is a must
358 if (responseCode >= 300 && responseCode <= 399 && !customHeaders.ContainsKey("Location"))
359 {
360 throw new InvalidRedirectRoute();
361 }
362 }
363
364 if (config.CustomGlobalHeaders.Any())
365 {
366 foreach (var h in config.CustomGlobalHeaders)
367 {
368 headers += $"{h.Key}: {h.Value}{NEW_LINE}";
369 }
370 }
371
372 if (config.CustomGlobalCookies.Any())
373 {
374 foreach (var c in config.CustomGlobalCookies)
375 {
376 headers += $"Set-Cookie: {c.Value}{NEW_LINE}";
377 }
378 }
379
380 /* if (request.GetHeaders["Connection"] != null)
381 {
382 headers += $"Connection: {request.GetHeaders["Connection"]}";
383 }
384 else
385 {*/
386 //visit https://httpwg.org/specs/rfc9113.html#ConnectionSpecific, p8.2.2 (27-Jun-23)
387 if (request.PROTOCOL == HTTP_PROTOCOL.HTTP1_0 || request.PROTOCOL == HTTP_PROTOCOL.HTTP1_1)
388 headers += "Connection: Close";
389
390 //}
391 headers += NEW_LINE + NEW_LINE;
392 return headers;
393 }
