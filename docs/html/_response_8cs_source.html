106 public void SendFile(string absPath, string? mimeType = null, int statusCode = HTTP_CODES.OK, Dictionary<string, string>? customHeaders = null)
107 {
108 var data = File.ReadAllBytes(absPath);
109
110 string _mime = mimeType ?? MimeTypeUtils.GetMimeType(Path.GetExtension(absPath)) ?? MimeTypeUtils.APPLICATION_OCTET;
111 string headers = GetHeaders(statusCode, data.Length, _mime, customHeaders);
112 byte[] headersBytes = Encoding.UTF8.GetBytes(headers);
113 byte[] responseBytes = new byte[data.Length + headersBytes.Length];
114
115 headersBytes.CopyTo(responseBytes, 0);
116 data.CopyTo(responseBytes, headersBytes.Length);
117
118 Send(responseBytes);
119 }

282 private string GetHeaders(int responseCode, int size, string contentType, Dictionary<string, string>? customHeaders = null)
283 {
284 CultureInfo ci = new("en-US");
285
286 string currentTime = DateTime.Now.ToString("ddd, dd MMM yyy HH:mm:ss ", ci) + "GMT";
287
288 string headers = $"{HttpUtils.ProtocolAsString(request.PROTOCOL)} {responseCode} {request.URL} {NEW_LINE}";
289 headers += "Date: " + currentTime + NEW_LINE;
290 if (config.CustomServerName != "")
291 headers += $"Server: {config.CustomServerName}{NEW_LINE}";
292 else
293 headers += $"Server: HSB-#/{Assembly.GetExecutingAssembly().GetName().Version} ({GetOSInfo()})" + NEW_LINE;
294
295 headers += $"Last-Modified: {currentTime}{NEW_LINE}";
296
297
298 if (responseCode < 300 || responseCode > 399)
299 {
300 headers += $"Content-Length: {size}{NEW_LINE}";
301 headers += $"Content-Type: {contentType}{NEW_LINE}";
302 }
303
304 if (customHeaders != null)
305 {
306
307 foreach (var h in customHeaders)
308 headers += $"{h.Key}: {h.Value}{NEW_LINE}";
309
310 //if it's a redirect "Location" header is a must
311 if (responseCode >= 300 && responseCode <= 399 && !customHeaders.ContainsKey("Location"))
312 {
313 throw new InvalidRedirectRoute();
314 }
315 }
316
317 if (config.CustomGlobalHeaders.Any())
318 {
319 foreach (var h in config.CustomGlobalHeaders)
320 {
321 headers += $"{h.Key}: {h.Value}{NEW_LINE}";
322 }
323 }
324
325 if (config.CustomGlobalCookies.Any())
326 {
327 foreach (var c in config.CustomGlobalCookies)
328 {
329 headers += $"Set-Cookie: {c.Value}{NEW_LINE}";
330 }
331 }
332
333 /* if (request.GetHeaders["Connection"] != null)
334 {
335 headers += $"Connection: {request.GetHeaders["Connection"]}";
336 }
337 else
338 {*/
339 //visit https://httpwg.org/specs/rfc9113.html#ConnectionSpecific, p8.2.2 (27-Jun-23)
340 if (request.PROTOCOL == HTTP_PROTOCOL.HTTP1_0 || request.PROTOCOL == HTTP_PROTOCOL.HTTP1_1)
341 headers += "Connection: Close";
342
343 //}
344 headers += NEW_LINE + NEW_LINE;
345 return headers;
346 }
