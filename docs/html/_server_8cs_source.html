125 public void Start(bool openInBrowser = false)
126 {
127
128 try
129 {
130 listener.Bind(localEndPoint);
131 listener.Listen(100);
132
133 var sslConf = config.SslSettings;
134
135 if (sslConf.IsEnabled)
136 { //sslListener and sslLocalEndPoint are not null because we checked in the constructor
137 sslListener!.Bind(sslLocalEndPoint!);
138 sslListener.Listen(100);
139 }
140
141 OpenInBrowserIfSet(openInBrowser, sslConf.IsEnabled, sslConf.PortMode == SSL_PORT_MODE.DUAL_PORT ? sslLocalEndPoint! : localEndPoint);
142
143 //this makes the second port listen to SSL requests
144 if (sslConf.IsEnabled && sslConf.PortMode == SSL_PORT_MODE.DUAL_PORT)
145 {
146 new Task(() =>
147 {
148 while (true)
149 {
150 Step(sslListener!, true);
151 }
152 }).Start();
153 }
154
155 //since the base port is alwyas listening this is always executed
156 while (true)
157 {
158 //if ssl is enabled and single port is used
159 var sslMode = sslConf.IsEnabled && sslConf.PortMode == SSL_PORT_MODE.SINGLE_PORT;
160
161 Step(listener, sslMode);
162 }
163 }
164 catch (Exception e)
165 {
166 Terminal.ERROR(e);
167 }

184 private void Step(Socket listener, bool sslMode)
185 {
186 Socket socket = listener.Accept();
187 byte[] bytes = new byte[config.RequestMaxSize];
188 int bytesRec = 0;
189 SslStream? sslStream = null;
190 if (sslMode)
191 {
192 var netstream = new NetworkStream(socket);
193 sslStream = new(netstream, true);
194 try
195 {
196 sslStream.AuthenticateAsServer(
197 config.SslSettings.GetCertificate(),
198 config.SslSettings.ClientCertificateRequired,
199 config.SslSettings.GetProtocols(),
200 config.SslSettings.CheckCertificateRevocation
201 );
202 bytesRec = sslStream.Read(bytes);
203 }
204 catch (Exception)
205 {
206 //if auth fails, the behaviour depends on the configuration
207 sslStream.Dispose();
208 if (config.SslSettings.upgradeUnsecureRequests)
209 {
210 config.Debug.WARNING("SSL authentication failed, redirecting to SSL");
211 //attempt redirect
212 Request _req = new(bytes, socket, config);
213 Response res = new(socket, _req, config, null);
214
215 //the endpoint varies if by the port mode
216 //if the port mode is dual port, we redirect to the ssl port
217 IPEndPoint redirectEndpoint = config.SslSettings.PortMode == SSL_PORT_MODE.DUAL_PORT ? sslLocalEndPoint! : localEndPoint;
218 res.Redirect("https://" + redirectEndpoint, HTTP_CODES.MOVED_PERMANENTLY);
219 return;
220 }
221
222 else
223 {
224 config.Debug.WARNING("SSL authentication failed, closing connection");
225 socket.Close();
226 return;
227 }
228 }
229 }
230 else
231 {
232 if(config.SslSettings.IsEnabled && config.SslSettings.upgradeUnsecureRequests){
233 config.Debug.WARNING("Unsecure request received, redirecting to SSL");
234 //attempt redirect
235 Request _req = new(bytes, socket, config);
236 Response res = new(socket, _req, config, null);
237
238 //the endpoint varies if by the port mode
239 //if the port mode is dual port, we redirect to the ssl port
240 IPEndPoint redirectEndpoint = config.SslSettings.PortMode == SSL_PORT_MODE.DUAL_PORT ? sslLocalEndPoint! : localEndPoint;
241 res.Redirect("https://" + redirectEndpoint, HTTP_CODES.MOVED_PERMANENTLY);
242 return;
243 }
244 bytesRec = socket.Receive(bytes);
245 }
246
247 bytes = bytes[..bytesRec]; //trim the array to the actual size of the request
248
249 //parse the request
250 Request req = new(bytes, socket, config);
251 //if is valid we process it
252 if (req.IsValidRequest)
253 {
254 Response res = new(socket, req, config, sslStream);
255 new Task(() => ProcessRequest(req, res)).Start();
256 }
257 else
258 {
259 //abort if the request is not valid
260 socket.Close();
261 }
262 }

279 protected internal static Dictionary<Tuple<string, bool>, Type> CollectStaticRoutes()
280 {
281 AppDomain currentDomain = AppDomain.CurrentDomain;
282 List<Assembly> assemblies = currentDomain.GetAssemblies().ToList();
283
284 assemblies.RemoveAll(a => a.ToString().StartsWith("System"));
285 assemblies.RemoveAll(a => a.ToString().StartsWith("Microsoft"));
286 assemblies.RemoveAll(a => a.ToString().StartsWith("Internal"));
287
288 Dictionary<Tuple<string, bool>, Type> routes = new();
289
290
291 foreach (var assem in assemblies)
292 {
293 List<Type> classes = assem.GetTypes().ToList();
294
295 foreach (var c in classes)
296 {
297 try
298 {
299 IEnumerable<Binding> multiBindings = c.GetCustomAttributes<Binding>(false);
300
301 //if no class have more than one binding, we search the ones with only one
302 if (multiBindings.Any())
303 {
304 foreach (Binding b in multiBindings)
305 {
306 if (b.Path != "")
307 routes.Add(new(b.Path, b.StartsWith), c);
308 }
309 }
310 else
311 {
312 Binding? attr = c.GetCustomAttribute<Binding>(false);
313 if (attr != null && attr.Path != "")
314 routes.Add(new(attr.Path, attr.StartsWith), c);
315 }
316 }
317 catch (Exception)
318 {
319 }
320 }
321 }
322
323 return routes;
324 }

339 private object? GetInstance(Request req, Response res)
340 {
341 Dictionary<Tuple<string, bool>, Type> routes = CollectStaticRoutes();
342
343
344 if (routes.ContainsKey(new(req.URL, false)))
345 {
346 Type c = routes[new(req.URL, false)];
347 var x = c.GetConstructors()[0];
348 return x.GetParameters().Length switch
349 {
350 3 => Activator.CreateInstance(c, req, res, config),
351 2 => Activator.CreateInstance(c, req, res),
352 _ => throw new Exception($"Invalid constructor found {x.Name}"),
353 };
354 }
355
356 //we omit the else branch
357 //we check if there is a a path that starts like the request url
358 return (from r in routes
359 let path = r.Key.Item1
360 where req.URL.StartsWith(path)
361 select r.Value
362 into c
363 let x = c.GetConstructors()[0]
364 select x.GetParameters().Length switch
365 {
366 3 => Activator.CreateInstance(c, req, res, config),
367 2 => Activator.CreateInstance(c, req, res),
368 _ => throw new Exception($"Invalid constructor found {x.Name}"),
369 }).FirstOrDefault();
370 }
376 private bool Filter(Request req)
377 {
378 var blockMode = config.BlockMode;
379 switch (blockMode)
380 {
381 case BLOCK_MODE.NONE: return false; //no blocking
382 case BLOCK_MODE.BANLIST:
383 if (config.PermanentIPList.Any(ip => ip == req.ClientIP))
384 {
385 return true; //blocked request
386 }
387 if (File.Exists("./banned_ips.txt"))
388 {
389 var bannedIps = File.ReadAllLines("./banned_ips.txt");
390 if (bannedIps.Contains(req.ClientIP))
391 {
392 return true; //blocked request
393 }
394 }
395 return true;
396 case BLOCK_MODE.OKLIST:
397 if (config.PermanentIPList.Any(ip => ip == req.ClientIP))
398 {
399 return false; //allowed request
400 }
401
402 if (File.Exists("./allowed_ips.txt"))
403 {
404 var allowedIps = File.ReadAllLines("./allowed_ips.txt");
405 return !allowedIps.Contains(req.ClientIP);
406 }
407
408 return false; //no allowed_ips.txt file found, so we allow all requests
409 }
410 return false;
411 }

414 private void ProcessRequest(Request req, Response res)
415 {
416 try
417 {
418 //check if request is valid
419 if (!req.validRequest)
420 {
421 config.Debug.WARNING($"{req.METHOD} '{req.URL}' {HTTP_CODES.NOT_FOUND} (Invalid Request)", true);
422 new Error(req, res, config, "Invalid Request", HTTP_CODES.NOT_FOUND).Process();
423 return;
424 }
425
426
427 //check if there is a filter that blocks the request
428 if (Filter(req)) return;
429
430 //check if server is launched with --listFiles
431 if (config.GetRawArguments().Contains("--listFiles"))
432 {
433 if (CheckSafePath(req.URL, req, res)) return;
434
435 new FileList(req, res, config).Process();
436 return;
437 }
438
439
440 //if dev has used the express mapping, we run the mapped function
441 if (RunIfExpressMapping(req, res)) return;
442
443 //We check if the route requested is handled by any servlet
444 object? o = GetInstance(req, res);
445
446 if (o != null)
447 {
448 //we check if the object is a servlet or a websocket
449 if (o is Servlet s)
450 {
451 s.Process();
452 }
453 else if (o is WebSocket ws)
454 {
455 if (!req.IsWebSocket())
456 {
457 config.Debug.WARNING($"{req.METHOD} '{req.URL}' {HTTP_CODES.METHOD_NOT_ALLOWED} (Invalid Request)", true);
458 new Error(req, res, config, "Invalid Request", HTTP_CODES.METHOD_NOT_ALLOWED).Process();
459 return;
460 }
461 else
462 {
463
464 ws.Process();
465 return;
466 }
467 }
468 else throw new Exception($"Developer tried to map an invalid object to a route -> {o.GetType()}");
469 }
470 else
471 {
472 //the client searched for a route that is not mapped by any servlet
473 //so we do some other checks like root page or static resource
474 //if no root page is set we search for and index.html file, else we show the default home page
475 if (req.URL == "/")
476 {
477 //if the client is requesting the root file, we check if there is an index.html file
478 //if not, we use the default servlet
479 if (File.Exists(config.StaticFolderPath + "/index.html"))
480 {
481 config.Debug.INFO($"{req.METHOD} '{req.URL}' 200");
482 res.SendHTMLFile(config.StaticFolderPath + "/index.html");
483 }
484 else
485 {
486 config.Debug.INFO($"{req.METHOD} '{req.URL}' 200 (Default Index Page)");
487 new Index(req, res, config).Process();
488 }
489 }
490 else
491 {
492 //we check if the client is requesting a resource, else 404 not found
493 //to check if the path is safe we use the same regex used in send.js
494 //see: https://github.com/pillarjs/send/blob/master/index.js#L63
495 if (CheckSafePath(req.URL, req, res))
496 {
497 return;
498 }
499 //if the path is safe, the static folder is set and the file exists, we send it
500 if (config.StaticFolderPath != "" && File.Exists(config.StaticFolderPath + "/" + req.URL))
501 {
502 //config.debug.INFO($"Static file found, serving '{req.URL}'");
503 config.Debug.INFO($"{req.METHOD} '{req.URL}' 200 (Static file)");
504 res.SendFile(config.StaticFolderPath + "/" + req.URL);
505 }
506 else if (config.ServeEmbeddedResource && Utils.IsEmbeddedResource(req.URL, config.EmbeddedResourcePrefix))
507 {
508 config.Debug.INFO($"{req.METHOD} '{req.URL}' 200 (Embedded resource)");
509 object resource = Utils.LoadResource<object>(req.URL, config.EmbeddedResourcePrefix);
510 res.SendObject(resource, req.URL);
511 }
512 else
513 {
514 //if no servlet or static file found, send 404
515 config.Debug.INFO($"{req.METHOD} '{req.URL}' 404 (Resource not found)");
516 new Error(req, res, config, "Page not found", HTTP_CODES.NOT_FOUND).Process();
517 }
518 }
519 }
520 }
521 catch (Exception e)
522 {
523 //config.debug.ERROR("Error handling request ->\n " + e);
524 config.Debug.ERROR($"{req.METHOD} '{req.URL}' 500 (Internal Server Error)\n{e}");
525 //we show an error page with the message and code 500
526 new Error(req, res, config, e.ToString(), HTTP_CODES.INTERNAL_SERVER_ERROR).Process();
527 }
528 }

531 {
532 if (Utils.IsUnsafePath(path))

535 new Error(req, res, config, "", HTTP_CODES.NOT_FOUND).Process();
536 if (config.IPAutoblock)
537 {
538 config.Debug.WARNING($"Autoblocking IP {req.ClientIP}");
539 if (File.Exists("./banned_ips.txt"))
540 File.AppendAllText("./banned_ips.txt", req.ClientIP + "\n");

543 }
544 return true;
545 }
546 return false;
