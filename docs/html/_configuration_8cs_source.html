131 public Configuration(string content)
132 {
133 using var doc = JsonDocument.Parse(content);
134 var root = doc.RootElement;
135
136 try
137 {
138 Address = root.GetProperty("Address").GetString() ?? "";
139 Port = root.GetProperty("Port").GetUInt16();
140 StaticFolderPath = root.GetProperty("StaticFolderPath").GetString() ?? "";
141 Debug = Debugger.FromJson(root.GetProperty("Debug"));
142 SslSettings = SslConfiguration.FromJSON(root.GetProperty("SslConfiguration"));
143 RequestMaxSize = root.GetProperty("Port").GetInt32();
144 BlockMode = (BLOCK_MODE)root.GetProperty("BlockMode").GetInt32();
145 HideBranding = root.GetProperty("HideBranding").GetBoolean();
146 IPAutoblock = root.GetProperty("IPAutoblock").GetBoolean();
147 ListeningMode = (IPMode)root.GetProperty(nameof(ListeningMode)).GetInt32();
148 CustomServerName = root.GetProperty("CustomServerName").GetString() ?? "";
149 ServeEmbeddedResource = root.GetProperty("ServeEmbeddedResource").GetBoolean();
150 EmbeddedResourcePrefix = root.GetProperty("EmbeddedResourcePrefix").GetString() ?? "";
151 DefaultSessionExpirationTime = root.GetProperty("DefaultSessionExpirationTime").GetUInt64();
152
153
154 foreach (var item in root.GetProperty("PermanentIPList").EnumerateArray())
155 {
156 string? v = item.GetString();
157 if (v != null)
158 {
159 PermanentIPList.Add(v!);
160 }
161 }
162
163
164
165 }
166 catch (Exception e)
167 {
168 Terminal.ERROR("Cannot parse configuration file");
169 Terminal.ERROR(e);
170 Environment.Exit(1);
171 }
172 }

353 public override string ToString()
354 {
355 string str = $"Current configuration:\nListening address and port: {Address}:{Port}";
356 if (StaticFolderPath == "")
357 str += "\nStatic folder is not set";
358 else
359 str += $"\nStatic folder path: {StaticFolderPath}";
360
361 if (expressMapping.Any())
362 {
363 str += "\nExpressJS-Like routing map:";
364 expressMapping.ForEach(m => str += $"\nPath : {m.Item1} -> {m.Item2.Item2.Method.Name}");
365 }
366
367 var staticRoutes = Server.CollectStaticRoutes();
368
369 if (staticRoutes.Any())
370 {
371 str += "\nStatic routes:";
372 staticRoutes.ToList().ForEach(m => str += $"\nPath : {m.Key.Item1} -> {m.Value.Name}");
373 }
374 return str;
375 }

377 public List<Tuple<string, string>> GetAllRoutes()
378 {
379 List<Tuple<string, string>> routes = new();
380 expressMapping.ForEach(m => routes.Add(new(m.Item1, m.Item2.Item1.ToString())));
381 var staticRoutes = Server.CollectStaticRoutes();
382 staticRoutes.ToList().ForEach(m => routes.Add(new(m.Key.Item1, "")));
383 return routes;
384 }
