93 Response res = new(socket, req, config);

119 protected internal static Dictionary<Tuple<string, bool>, Type> CollectStaticRoutes()
120 {
121 AppDomain currentDomain = AppDomain.CurrentDomain;
122 List<Assembly> assemblies = currentDomain.GetAssemblies().ToList();
123
124 assemblies.RemoveAll(a => a.ManifestModule.Name.StartsWith("System"));
125 assemblies.RemoveAll(a => a.ManifestModule.Name.StartsWith("Microsoft"));
126 assemblies.RemoveAll(a => a.ManifestModule.Name.StartsWith("Internal"));
127
128 Dictionary<Tuple<string, bool>, Type> routes = new();
129
130
131 foreach (var assem in assemblies)
132 {
133 List<Type> classes = assem.GetTypes().ToList();
134
135 foreach (var c in classes)
136 {
137 try
138 {
139 IEnumerable<Binding> multiBindings = c.GetCustomAttributes<Binding>(false);
140
141 //if no class have more than one binding, we search the ones with only one
142 if (multiBindings.Any())
143 {
144 foreach (Binding b in multiBindings)
145 {
146 if (b.Path != "")
147 routes.Add(new(b.Path, b.StartsWith), c);
148 }
149 }
150 else
151 {
152 Binding? attr = c.GetCustomAttribute<Binding>(false);
153 if (attr != null && attr.Path != "")
154 routes.Add(new(attr.Path, attr.StartsWith), c);
155 }
156 }
157 catch (Exception)
158 {
159 }
160 }
161 }
162
163 return routes;
164 }

179 private object? GetInstance(Request req, Response res)
180 {
181 Dictionary<Tuple<string, bool>, Type> routes = CollectStaticRoutes();
182
183
184 if (routes.ContainsKey(new(req.URL, false)))
185 {
186 Type c = routes[new(req.URL, false)];
187 var x = c.GetConstructors()[0];
188 return x.GetParameters().Length switch
189 {
190 3 => Activator.CreateInstance(c, req, res, config),
191 2 => Activator.CreateInstance(c, req, res),
192 _ => throw new Exception($"Invalid constructor found {x.Name}"),
193 };
194 }
195
196 //we omit the else branch
197 //we check if there is a a path that starts like the request url
198 return (from r in routes
199 let path = r.Key.Item1
200 where req.URL.StartsWith(path)
201 select r.Value
202 into c
203 let x = c.GetConstructors()[0]
204 select x.GetParameters().Length switch
205 {
206 3 => Activator.CreateInstance(c, req, res, config),
207 2 => Activator.CreateInstance(c, req, res),
208 _ => throw new Exception($"Invalid constructor found {x.Name}"),
209 }).FirstOrDefault();
210 }

212 private void ProcessRequest(Request req, Response res)
213 {
214 try
215 {
216 //check if request is valid
217 if (!req.validRequest)
218 {
219 config.debug.WARNING($"{req.METHOD} '{req.URL}' {HTTP_CODES.NOT_FOUND} (Invalid Request)", true);
220 new Error(req, res, config, "Invalid Request", HTTP_CODES.NOT_FOUND).Process();
221 return;
222 }
223 if (RunIfExpressMapping(req, res))
224 return;
225
226 //We check if the route requested is handled by any servlet
227 object? o = GetInstance(req, res);
228
229 if (o != null)
230 {
231 //we check if the object is a servlet or a websocket
232 if (o is Servlet s)
233 {
234 s.Process();
235 }
236 else if (o is WebSocket ws)
237 {
238 if(!req.IsWebSocket())
239 {
240 config.debug.WARNING($"{req.METHOD} '{req.URL}' {HTTP_CODES.METHOD_NOT_ALLOWED} (Invalid Request)", true);
241 new Error(req, res, config, "Invalid Request", HTTP_CODES.METHOD_NOT_ALLOWED).Process();
242 return;
243 }
244 else{
245
246 ws.Process();
247 return;
248 }
249 }
250 else throw new Exception($"Developer tried to map an invalid object to a route -> {o.GetType()}");
251 }
252 else
253 {
254 //the client searched for a route that is not mapped by any servlet
255 //so we do some other checks like root page or static resource
256 //if no root page is set we search for and index.html file, else we show the default home page
257 if (req.URL == "/")
258 {
259 //if the client is requesting the root file, we check if there is an index.html file
260 //if not, we use the default servlet
261 if (File.Exists(config.staticFolderPath + "/index.html"))
262 {
263 res.SendHTMLFile(config.staticFolderPath + "/index.html");
264 config.debug.INFO($"{req.METHOD} '{req.URL}' 200");
265 }
266 else
267 {
268 new Index(req, res, config).Process();
269 config.debug.INFO($"{req.METHOD} '{req.URL}' 200 (Default Index Page)");
270 }
271 }
272 else
273 {
274 //we check if the client is requesting a resource, else 404 not found
275 //to check if the path is safe we use the same regex used in send.js
276 //see: https://github.com/pillarjs/send/blob/master/index.js#L63
277 if (Utils.IsUnsafePath(req.URL))
278 {
279 config.debug.WARNING($"{req.METHOD} '{req.URL}' 200 (Requested unsafe path, ignoring request)");
280 new Error(req, res, config, "", HTTP_CODES.NOT_FOUND).Process();
281 }
282 //if the path is safe and the file exists, we send it
283 if (File.Exists(config.staticFolderPath + "/" + req.URL))
284 {
285 //config.debug.INFO($"Static file found, serving '{req.URL}'");
286 config.debug.INFO($"{req.METHOD} '{req.URL}' 200 (Static file)");
287 res.SendFile(config.staticFolderPath + "/" + req.URL);
288 }
289 else
290 {
291 //if no servlet or static file found, send 404
292 config.debug.INFO($"{req.METHOD} '{req.URL}' 404 (Resource not found)");
293 new Error(req, res, config, "Page not found", HTTP_CODES.NOT_FOUND).Process();
294 }
295 }
296 }
297 }
298 catch (Exception e)
299 {
300 //config.debug.ERROR("Error handling request ->\n " + e);
301 config.debug.ERROR($"{req.METHOD} '{req.URL}' 500 (Internal Server Error)\n{e}");
302 //we show an error page with the message and code 500
303 new Error(req, res, config, e.ToString(), HTTP_CODES.INTERNAL_SERVER_ERROR).Process();
304 }
305 }
