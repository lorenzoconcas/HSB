280 public override string ToString()
281 {
282 string str = $"Current configuration:\nListening address and port: {address}:{port}";
283 if (staticFolderPath == "")
284 str += "\nStatic folder is not set";
285 else
286 str += $"\nStatic folder path: {staticFolderPath}";
287
288 if (expressMapping.Any())
289 {
290 str += "\nExpressJS-Like routing map:";
291 expressMapping.ForEach(m => str += $"\nPath : {m.Item1} -> {m.Item2.Item2.Method.Name}");
292 }
293
294 var staticRoutes = Server.CollectStaticRoutes();
295
296 if (staticRoutes.Any())
297 {
298 str += "\nStatic routes:";
299 staticRoutes.ToList().ForEach(m => str += $"\nPath : {m.Key.Item1} -> {m.Value.Name}");
300 }
301 return str;
302 }
