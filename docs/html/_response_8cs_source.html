103 public void SendFile(string absPath, string? mimeType = null, int statusCode = 200, Dictionary<string, string>? customHeaders = null)
104 {

107 string _mime = mimeType ?? MimeType.GetMimeType(Path.GetExtension(absPath));

250 private string GetHeaders(int responseCode, int size, string contentType, Dictionary<string, string>? customHeaders = null)
251 {
252 CultureInfo ci = new("en-US");
253
254 string currentTime = DateTime.Now.ToString("ddd, dd MMM yyy HH:mm:ss GMT", ci);
255
256 string headers = $"{HttpUtils.ProtocolAsString(request.PROTOCOL)} {responseCode} {request.URL} {NEW_LINE}";
257 headers += "Date: " + currentTime + NEW_LINE;
258
259 headers += $"Server : HSB-#/{Assembly.GetExecutingAssembly().GetName().Version} ({GetOSInfo()})" + NEW_LINE;
260 headers += $"Last-Modified: {currentTime}{NEW_LINE}";
261 headers += $"Content-Length: {size}{NEW_LINE}";
262 headers += $"Content-Type: {contentType}{NEW_LINE}";
263
264 if (customHeaders != null)
265 foreach (var h in customHeaders)
266 headers += $"{h.Key}: {h.Value}{NEW_LINE}";
267
268 if (config.CustomGlobalHeaders.Any())
269 {
270 foreach (var h in config.CustomGlobalHeaders)
271 {
272 headers += $"{h.Key}: {h.Value}{NEW_LINE}";
273 }
274 }
275
276 if (config.CustomGlobalCookies.Any())
277 {
278 foreach (var c in config.CustomGlobalCookies)
279 {
280 headers += $"Set-Cookie: {c.Value}{NEW_LINE}";
281 }
282 }
